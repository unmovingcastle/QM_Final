\section{Momentum}

\subsection{velocity and momentum}
We derive the expressions for $\expval{v}$ as well as $\expval{p}$ in this section. From Eq. \tooltip{\ref{eq:1_4_4}}{\parbox{10cm}{
\begin{equation*}
    \pdv{}{t}\abs{\Psi}^2  = \pdv{}{x}\qty[\frac{i\hbar}{2m}\qty(\Psi^*\pdv[]{\Psi}{x}-\pdv[]{\Psi^*}{x}\Psi)]
\end{equation*}}} as well as the fact that 
\begin{equation*}
    \expval{x}=\int_{-\infty}^{\infty}x\abs{\Psi(x,t)}^2\dd x,
\end{equation*} 
it follows that 
\begin{align*}
    \dv{\expval{x}}{t}=\dv{t}\int_{-\infty}^{\infty}x\abs{\Psi(x,t)}^2\dd x = \int_{-\infty}^{\infty}\pdv{t}\qty(x\abs{\Psi(x,t)}^2)\dd x =\int_{-\infty}^{\infty}x\pdv{t}\abs{\Psi(x,t)}^2\dd x
\end{align*}
($x$ can be taken out of the partial derivative, since we are differentiating with respect to $t$), substituting in the expression:
\begin{align*}
    \int_{-\infty}^\infty x\pdv{t}\abs{\Psi(x,t)}^2\dd x &= \int_{-\infty}^\infty x\pdv{}{x}\qty[\frac{i\hbar}{2m}\qty(\Psi^*\pdv[]{\Psi}{x}-\pdv[]{\Psi^*}{x}\Psi)]\dd x\\
    &= \frac{i\hbar}{2m}\int_{-\infty}^\infty x\pdv{}{x}\qty(\Psi^*\pdv[]{\Psi}{x}-\pdv[]{\Psi^*}{x}\Psi)\dd x
\end{align*}
Integration-by-parts leads to
\begin{align*}
    -\frac{i\hbar}{2m}\int_{-\infty}^\infty \dv{x}{x}\qty(\Psi^*\pdv[]{\Psi}{x}-\pdv[]{\Psi^*}{x}\Psi)\dd x + x\qty(\Psi^*\pdv[]{\Psi}{x}-\pdv[]{\Psi^*}{x}\Psi)\eval_{-\infty}^{\infty}
\end{align*}
where the last term on the right, again, goes to zero due to the wave function's being normalizable. We are now left with just the integral on the left. Integrate by parts again to obtain
\begin{align*}
    -\frac{i\hbar}{2m}\int_{-\infty}^\infty \dv{x}{x}\qty(\Psi^*\pdv[]{\Psi}{x}-\pdv[]{\Psi^*}{x}\Psi)\dd x 
    &= -\frac{i\hbar}{2m}\int_{-\infty}^\infty \Psi^* \pdv{\Psi}{x}-\pdv{\Psi^*}{x}\Psi \dd x \\
    &= \frac{i\hbar}{2m}\qty(\int_{-\infty}^\infty -\Psi^* \pdv{\Psi}{x}\dd x+\int_{-\infty}^\infty \Psi \pdv{\Psi^*}{x}\dd x) \\
     &= \frac{i\hbar}{2m}\qty(\int_{-\infty}^\infty -\Psi^* \pdv{\Psi}{x}\dd x+\int_{-\infty}^\infty -\Psi^* \pdv{\Psi}{x}\dd x+\Psi\Psi^*\eval_{-\infty}^{\infty}) \\
     &= \frac{i\hbar}{2m}\qty(\int_{-\infty}^\infty -2\Psi^* \pdv{\Psi}{x}\dd x)\\
     &= \frac{-i\hbar}{m}\qty(\int_{-\infty}^\infty \Psi^* \pdv{\Psi}{x}\dd x)
\end{align*}
One again, $\Psi(x,t)$ goes to zero as $x$ approaches $\pm$ infinity. Hence, we arrive at 
\begin{equation}
    \dv{\expval{x}}{t}=-\frac{i\hbar}{m}\int\Psi^*\pdv{\Psi}{x}\dd x
\end{equation}
That is,
\begin{equation}
    \expval{v}=\dv[]{\expval{x}}{t}=-\frac{i\hbar}{m}\int\Psi^*\pdv{\Psi}{x}\dd x
\end{equation}
It is a well known fact that the momentum is $p = mv$. It is only natural that
\begin{align}
    \expval{p} = m\expval{v} = -i\hbar\int\Psi^*\pdv{\Psi}{x}\dd x
    \label{eq:1_5_3}
\end{align}

\subsection{Ehrenfest's theorem (problem 7)}
Take the derivative of Equation \ref{eq:1_5_3} with respect to time to obtain
\begin{equation*}
    \setlength{\jot}{13pt}
    \begin{aligned}
    \dv[]{\expval{p}}{t} &= -i\hbar\int_{-\infty}^{\infty} \pdv{t}\qty(\Psi^*\pdv{\Psi}{x})\dd x\\
    &=-i\hbar\int_{-\infty}^{\infty} \pdv{\Psi^*}{t}\pdv{\Psi}{x} + \Psi^*\pdv[2]{\Psi}{t}{x}\dd x
    \end{aligned}
\end{equation*}
Now, from Equations \ref{eq:1_4_2} and \ref{eq:1_4_3}, we have 
\begin{align*}
    \pdv{\Psi}{t} = \frac{i\hbar}{2m}\pdv[2]{\Psi}{x}-\frac{i}{\hbar}V\Psi ~~~~\textrm{and}~~~~
    \pdv{\Psi^*}{t} = -\frac{i\hbar}{2m}\pdv[2]{\Psi^*}{x}+\frac{i}{\hbar}V\Psi^*
\end{align*}
Thus,
\begin{equation*}
    \dv[]{\expval{p}}{t} = -i\hbar \int_{-\infty}^{\infty}\qty(\frac{-i\hbar}{2m}\pdv[2]{\Psi^*}{x}+\frac{i}{\hbar}V\Psi^*)\pdv{\Psi}{x}
    + \Psi^*\pdv{x}\qty(\frac{i\hbar}{2m}\pdv[2]{\Psi}{x}-\frac{i}{\hbar}V\Psi)\dd x 
\end{equation*}
Combine like terms to get
\begin{equation*}
    -i\hbar \int_{-\infty}^{\infty}
    \frac{i\hbar}{2m}\qty(\Psi^*\pdv[3]{\Psi}{x} - \pdv[2]{\Psi^*}{x}\pdv{\Psi}{x}) + \frac{i}{\hbar}\qty[V\Psi^*\pdv{\Psi}{x}-\Psi^*\pdv{x}(V\Psi)]\dd x 
\end{equation*}
The first term in the integral above is zero (integrate by parts twice, throw away the boundary terms), as shown \tooltip{here}{
\parbox{10cm}{
\begin{align*}
    \int\pdv[2]{\Psi^*}{x}\pdv{\Psi}{x}\dd x 
    &=-\int\pdv{\Psi^*}{x}\pdv[2]{\Psi}{x}\dd x + \pdv{\Psi}{x}\pdv{\Psi^*}{x}\Big|_{-\infty}^\infty \\
    &=-\int\pdv{\Psi^*}{x}\pdv[2]{\Psi}{x}\dd x + 0\\
    &= \int \Psi^*\pdv[3]{\Psi}{x}\dd x + \Psi^*\pdv[2]{\Psi}{x}\Big|_{-\infty}^{\infty}\\
    &= \int \Psi^*\pdv[3]{\Psi}{x}\dd x + 0
\end{align*}
\begin{align*}
    \therefore \int_{-\infty}^\infty \frac{i\hbar}{2m}\qty(\Psi^*\pdv[3]{\Psi}{x} -\pdv[2]{\Psi^*}{x}\pdv{\Psi}{x})\dd x = 0
\end{align*}
}
}, and so we are left with
\begin{equation*}
    \setlength{\jot}{13pt}
    \begin{aligned}
    -i\hbar \int_{-\infty}^{\infty}
     \frac{i}{\hbar}\qty[V\Psi^*\pdv{\Psi}{x}-\Psi^*\pdv{x}(V\Psi)]\dd x  &= -i\hbar \int_{-\infty}^{\infty} \frac{i}{\hbar}\qty[V\Psi^*\pdv{\Psi}{x}-\Psi^*\pdv{V}{x}\Psi-\Psi^*V\pdv{\Psi}{x }] \dd x\\
     &= -i\hbar\int_{-\infty}^\infty\frac{-i}{\hbar}\Psi^*\pdv{V}{x}\Psi \dd x
     = -i\hbar\int_{-\infty}^\infty\frac{-i}{\hbar}\abs{\Psi}^2\pdv{V}{x} \dd x\\
     &= -\int_{-\infty}^\infty\abs{\Psi}^2\pdv{V}{x} \dd x = \expval{-\pdv{V}{x}} ~\qed
    \end{aligned}
\end{equation*}
